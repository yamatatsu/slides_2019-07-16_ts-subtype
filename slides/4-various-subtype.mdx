## Readonlyのsubtype

---

<Editor code={`\
{ // file
  type A = { a: number }
  type AReadonly = Readonly<A>
\n\
  const aReadonly: AReadonly = { a: 1 }
  // aReadonly.a = 10 // error!
\n\
  mutate(aReadonly)
\n\
  console.log(aReadonly) // => { a: 100 }
\n\
  // 破壊
  function mutate(arg: A) {
    arg.a = 100
  }
}
// https://github.com/microsoft/TypeScript/issues/13347
`} />
